<script>

function analyzeCareer() {

let motivation = document.getElementById("motivation").value;
let technology = document.getElementById("technology").value;
let task = document.getElementById("task").value;

let career = "";
let confidence = 0;

if (task.includes("code") && technology.includes("Very")) {
    career = "Software Engineer / AI Developer";
    confidence = 92;
}
else if (task.includes("Designing")) {
    career = "UI/UX Designer";
    confidence = 85;
}
else if (task.includes("financial")) {
    career = "Business Analyst / Commerce Field";
    confidence = 88;
}
else if (task.includes("teaching")) {
    career = "Teacher / Psychologist";
    confidence = 90;
}
else if (task.includes("Managing")) {
    career = "Entrepreneur / Manager";
    confidence = 84;
}
else {
    career = "Medical Field / Explore Multiple Fields";
    confidence = 70;
}

let confidenceClass = "";

if (confidence >= 85) {
    confidenceClass = "high";
}
else if (confidence >= 70) {
    confidenceClass = "medium";
}
else {
    confidenceClass = "low";
}

document.getElementById("result").innerHTML =
"✨ Recommended Career: " + career +
"<br><br>📊 Confidence Level: <span class='" + confidenceClass + "'>" +
confidence + "%</span>" +
"<br><br>🤖 AI Explanation: Based on your choices, our AI logic matched your interests and skill comfort level with the most suitable career path.";

}

function checkQuiz() {

let score = 0;

let answers = ["q1","q2","q3","q4","q5"];

for (let i = 0; i < answers.length; i++) {
    let selected = document.querySelector('input[name="'+answers[i]+'"]:checked');
    if (selected && selected.value == "1") {
        score++;
    }
}

let percentage = (score/5) * 100;

document.getElementById("quizResult").innerHTML =
"🎉 Your Quiz Score: " + score + "/5<br>📊 Percentage: " + percentage + "%";

}
